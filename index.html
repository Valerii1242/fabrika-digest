import sqlite3 con = sqlite3.connect('data/scout_mvp.sqlite') con.row_factory = sqlite3.Row # колонки publication_drafts cols = [r['name'] for r in con.execute('PRAGMA table_info(publication_drafts)').fetchall()] print('DRAFT COLS:', cols) # все таблицы tables = [t[0] for t in con.execute("SELECT name FROM sqlite_master WHERE type='table'").fetchall()] print('TABLES:', tables) # сигнальная таблица sig = 'curated_signals' if 'curated_signals' in tables else 'signals' print('SIG TABLE:', sig) # последние 3 drafts print('\n--- LAST 3 DRAFTS ---') for r in con.execute('SELECT * FROM publication_drafts ORDER BY created_at DESC LIMIT 3').fetchall(): for k in r.keys(): print(' ', k, ':', str(r[k])[:100]) print(' ---') # последние 3 runs print('\n--- LAST 3 RUNS ---') for r in con.execute('SELECT run_id, started_at, status, total, taken FROM scout_runs ORDER BY started_at DESC LIMIT 3').fetchall(): print(' ', dict(r)) # все упоминания publication_drafts в app.py print('\n--- APP.PY publication_drafts mentions ---') txt = open('app.py', encoding='utf-8').read() lines = txt.split('\n') for i, line in enumerate(lines): if 'publication_drafts' in line or 'get_drafts' in line or 'build_ai_review' in line: print(f' L{i+1}: {line.strip()[:120]}') # цепочка draft->signal->run print('\n--- DRAFT->SIGNAL->RUN chain ---') try: q = f'SELECT d.draft_id, d.signal_id, s.run_id, s.curator_status, d.created_at FROM publication_drafts d LEFT JOIN {sig} s ON d.signal_id = s.signal_id ORDER BY d.created_at DESC LIMIT 5' for r in con.execute(q).fetchall(): print(' draft:', str(r[0])[:8], '| sig:', str(r[1])[:8], '| run:', r[2], '| status:', r[3], '| at:', r[4]) except Exception as e: print(' ERROR:', e) con.close() print('\nDONE')
